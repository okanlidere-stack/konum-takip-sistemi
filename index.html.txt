// ══════════════════════════════════════════════════

let watchId        = null;
let isTracking     = false;
let sessions       = [];
let currentSession = null;
let timerInterval  = null;
let trackStartTime = null;
let geocodeCache   = {};
const MOVE_THRESHOLD = 50;
const STORAGE_KEY    = 'konum_takip_v2';

// ══════════════════════════════════════════════════
//  PERSISTENCE
// ══════════════════════════════════════════════════
function saveState() {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify({
      sessions, currentSession, geocodeCache, trackStartTime, isTracking
    }));
  } catch(e) {}
}

function loadState() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return;
    const p = JSON.parse(raw);
    sessions       = p.sessions      || [];
    geocodeCache   = p.geocodeCache  || {};
    trackStartTime = p.trackStartTime || null;
    if (p.currentSession) {
      const cs = p.currentSession;
      if (p.isTracking) {
        cs.endTime = cs.endTime || Date.now();
        sessions.push(cs);
      } else {
        currentSession = cs;
      }
    }
  } catch(e) {}
}

// ══════════════════════════════════════════════════
//  REVERSE GEOCODING  (Nominatim, rate-limited 1/s)
// ══════════════════════════════════════════════════
const geocodeQueue = [];
let geocoding = false;

function reverseGeocode(lat, lng) {
  const key = `${lat.toFixed(4)},${lng.toFixed(4)}`;
  if (geocodeCache[key]) return Promise.resolve(geocodeCache[key]);
  return new Promise(resolve => {
    geocodeQueue.push({ lat, lng, key, resolve });
    if (!geocoding) processGeoQueue();
  });
}

async function processGeoQueue() {
  if (geocodeQueue.length === 0) { geocoding = false; return; }
  geocoding = true;
  const { lat, lng, key, resolve } = geocodeQueue.shift();
  try {
    const res  = await fetch(
      `https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lng}&format=json&addressdetails=1`,
      { headers: { 'Accept-Language': 'tr,en' } }
    );
    const data = await res.json();
    const a    = data.address || {};
    const parts = [];
    const road = a.road || a.pedestrian || a.footway || a.street || '';
    const num  = a.house_number || '';
    if (road) parts.push(num ? `${road} No:${num}` : road);
    const nb = a.neighbourhood || a.suburb || a.quarter || '';
    if (nb) parts.push(nb);
    const dist = a.district || a.city_district || a.town || '';
    if (dist && dist !== nb) parts.push(dist);
    const city = a.city || a.municipality || a.county || '';
    if (city && city !== dist) parts.push(city);
    const address = parts.length ? parts.join(', ') : (data.display_name || `${lat.toFixed(4)}, ${lng.toFixed(4)}`);
    geocodeCache[key] = address;
    saveState();
    resolve(address);
  } catch(e) {
    const fb = `${lat.toFixed(4)}°, ${lng.toFixed(4)}°`;
    geocodeCache[key] = fb;
    resolve(fb);
  }
  setTimeout(processGeoQueue, 1100);
}

// ══════════════════════════════════════════════════

function haversine(lat1,lon1,lat2,lon2) {
  const R=6371000, r=Math.PI/180;
  const φ1=lat1*r, φ2=lat2*r, Δφ=(lat2-lat1)*r, Δλ=(lon2-lon1)*r;
  const a=Math.sin(Δφ/2)**2+Math.cos(φ1)*Math.cos(φ2)*Math.sin(Δλ/2)**2;
  return R*2*Math.atan2(Math.sqrt(a),Math.sqrt(1-a));
}

function formatDuration(ms) {
  if (!ms||ms<0) return '0d 00s';
  const s=Math.floor(ms/1000), h=Math.floor(s/3600), m=Math.floor((s%3600)/60), sec=s%60;
  if (h>0) return `${h}s ${String(m).padStart(2,'0')}d`;
  return `${m}d ${String(sec).padStart(2,'0')}s`;
}

function formatTime(ts) {
  if (!ts) return '—';
  return new Date(ts).toLocaleTimeString('tr-TR',{hour:'2-digit',minute:'2-digit',second:'2-digit'});
}

function coordKey(lat,lng) {
  return `${(Math.round(lat*1000)/1000).toFixed(3)},${(Math.round(lng*1000)/1000).toFixed(3)}`;
}

// ══════════════════════════════════════════════════
//  TRACKING
// ══════════════════════════════════════════════════
function startTracking() {
  if (!navigator.geolocation) { showError('Tarayıcınız konum servisini desteklemiyor.'); return; }
  document.getElementById('startBtn').disabled = true;
  document.getElementById('stopBtn').disabled  = false;
  document.getElementById('currentPanel').classList.add('visible');
  document.getElementById('bgNotice').classList.add('show');
  setStatus('','Konum bekleniyor...');
  isTracking     = true;
  trackStartTime = trackStartTime || Date.now();
  saveState();
  const opts = { enableHighAccuracy:true, timeout:20000, maximumAge:5000 };
  watchId = navigator.geolocation.watchPosition(onPosition, onGeoError, opts);
  timerInterval = setInterval(() => {
    document.getElementById('totalTimeVal').textContent = formatDuration(Date.now()-trackStartTime);
    if (currentSession) renderList();
  }, 1000);
}

function stopTracking() {
  if (watchId!==null) navigator.geolocation.clearWatch(watchId);
  watchId=null; isTracking=false; clearInterval(timerInterval);
  if (currentSession) {
    currentSession.endTime=Date.now();
    sessions.push({...currentSession});
    currentSession=null;
  }
  saveState();
  setStatus('','Durduruldu');
  document.getElementById('startBtn').disabled=false;
  document.getElementById('stopBtn').disabled=true;
  document.getElementById('bgNotice').classList.remove('show');
  renderList(); renderReport();
  showInfo(`Takip durduruldu — ${sessions.length} oturum kaydedildi.`);
}

async function onPosition(pos) {
  const {latitude:lat, longitude:lng, accuracy:acc} = pos.coords;
  const now = Date.now();
  setStatus('active','Takip Aktif');
  document.getElementById('curLat').textContent  = lat.toFixed(5)+'°';
  document.getElementById('curLng').textContent  = lng.toFixed(5)+'°';
  document.getElementById('curAcc').textContent  = Math.round(acc)+' m';
  document.getElementById('curTime').textContent = formatTime(now);
  document.getElementById('accuracy').textContent= Math.round(acc)+'m';

  const curAddrEl = document.getElementById('curAddress');
  curAddrEl.className='current-address loading';
  curAddrEl.textContent='Adres çözümleniyor...';
  reverseGeocode(lat,lng).then(addr=>{
    curAddrEl.className='current-address';
    curAddrEl.textContent=addr;
  });

  if (!currentSession) {
    currentSession={lat,lng,acc,startTime:now,endTime:null,address:null};
    reverseGeocode(lat,lng).then(addr=>{
      currentSession.address=addr; saveState(); renderList(); updateCounts();
    });
    renderList(); return;
  }

  const dist = haversine(currentSession.lat,currentSession.lng,lat,lng);
  if (dist>MOVE_THRESHOLD) {
    currentSession.endTime=now;
    sessions.push({...currentSession});
    currentSession={lat,lng,acc,startTime:now,endTime:null,address:null};
    reverseGeocode(lat,lng).then(addr=>{
      currentSession.address=addr; saveState(); renderList(); renderReport();
    });
    saveState(); renderList(); renderReport(); updateCounts();
    showInfo(`Yeni konum — ${Math.round(dist)}m hareket`);
  }
}

function onGeoError(err) {
  setStatus('error','Hata');
  const msgs={1:'Konum izni reddedildi. Adres çubuğundaki kilit ikonundan izin verin.',
              2:'Konum alınamadı — GPS sinyali zayıf.',
              3:'Konum zaman aşımına uğradı.'};
  showError(msgs[err.code]||'Hata: '+err.message);
}

function updateCounts() {
  const all=[...sessions,...(currentSession?[currentSession]:[])];
  const keys=new Set(all.map(s=>coordKey(s.lat,s.lng)));
  document.getElementById('locationCount').textContent=keys.size;
}

// ══════════════════════════════════════════════════
//  RENDER LIST
// ══════════════════════════════════════════════════
function renderList() {
  const list=document.getElementById('locationList');
  const empty=document.getElementById('emptyState');
  const all=[...sessions,...(currentSession?[{...currentSession,isCurrent:true}]:[])];
  document.getElementById('entryCount').textContent=all.length+' kayıt';
  list.querySelectorAll('.location-entry').forEach(e=>e.remove());
  if (all.length===0){empty.style.display='block';return;}
  empty.style.display='none';
  [...all].reverse().forEach((s,ri)=>{
    const i=all.length-1-ri;
    const dur=s.isCurrent?(Date.now()-s.startTime):(s.endTime-s.startTime);
    const mapsUrl=`https://www.google.com/maps?q=${s.lat},${s.lng}`;
    const addr=s.address;
    const div=document.createElement('div');
    div.className='location-entry'+(s.isCurrent?' is-current':'');
    div.innerHTML=`
      <div class="entry-index">${String(i+1).padStart(2,'0')}</div>
      <div class="entry-main">
        <div class="entry-address ${addr?'':'resolving'}">${
          s.isCurrent
            ? (addr?'📍 '+addr:'📍 Adres çözümleniyor...')
            : (addr||'Adres yükleniyor...')
        }</div>
        <div class="entry-meta">
          <span class="hi">${s.lat.toFixed(4)}, ${s.lng.toFixed(4)}</span>
          <span>±${Math.round(s.acc)}m</span>
          <span>Giriş: ${formatTime(s.startTime)}</span>
          ${!s.isCurrent?`<span>Çıkış: ${formatTime(s.endTime)}</span>`:''}
        </div>

      </div>
      <div class="entry-duration">
        <div class="duration-value">${formatDuration(dur)}</div>
        <div class="duration-label">${s.isCurrent?'devam ediyor':'geçirildi'}</div>
      </div>`;
    list.appendChild(div);
  });
}

// ══════════════════════════════════════════════════
//  RENDER REPORT
// ══════════════════════════════════════════════════
function renderReport() {
  const completed=sessions.filter(s=>s.endTime);
  if (completed.length===0){document.getElementById('reportSection').style.display='none';return;}
  document.getElementById('reportSection').style.display='block';

  // Group by coord key
  const groups={};
  completed.forEach(s=>{
    const key=coordKey(s.lat,s.lng);
    if (!groups[key]) groups[key]={lat:s.lat,lng:s.lng,totalMs:0,visits:0,address:s.address};
    groups[key].totalMs+=(s.endTime-s.startTime);
    groups[key].visits++;
    if (s.address&&!groups[key].address) groups[key].address=s.address;
  });

  const sorted=Object.values(groups).sort((a,b)=>b.totalMs-a.totalMs);
  const grand =sorted.reduce((sum,g)=>sum+g.totalMs,0)||1;
  const tbody=document.getElementById('reportBody');
  tbody.innerHTML='';

  sorted.forEach((g,i)=>{
    const pct    =((g.totalMs/grand)*100).toFixed(1);
    const barPct =Math.round((g.totalMs/grand)*100);
    const addrStr=g.address||`${g.lat.toFixed(4)}, ${g.lng.toFixed(4)}`;
    const parts  =addrStr.split(', ');
    const prim   =parts.slice(0,2).join(', ');
    const sec    =parts.slice(2).join(', ');
    const tr=document.createElement('tr');
    tr.innerHTML=`
      <td style="font-family:'Space Mono',monospace;font-size:0.72rem;color:var(--muted)">${i+1}</td>
      <td>
        <div class="addr-primary">${prim}</div>
        ${sec?`<div class="addr-secondary">${sec}</div>`:''}
        <a class="map-link" href="https://www.google.com/maps?q=${g.lat},${g.lng}" target="_blank">↗ Harita</a>
      </td>
      <td style="font-family:'Space Mono',monospace;font-weight:700;color:var(--accent3);white-space:nowrap">${formatDuration(g.totalMs)}</td>
      <td style="font-family:'Space Mono',monospace;font-size:0.78rem;color:var(--muted)">${g.visits}x</td>
      <td class="pct-cell">
        <div class="pct-row">
          <div class="pct-num">%${pct}</div>
          <div class="bar-bg"><div class="bar-fill" style="width:${barPct}%"></div></div>
        </div>
      </td>`;
    tbody.appendChild(tr);
  });
}

// ══════════════════════════════════════════════════
//  CSV EXPORT
// ══════════════════════════════════════════════════
function exportCSV() {
  const done=sessions.filter(s=>s.endTime);
  if (!done.length){showError('Dışa aktarılacak tamamlanmış kayıt yok.');return;}
  const grand=done.reduce((sum,s)=>sum+(s.endTime-s.startTime),0)||1;
  let csv='No,Adres,Enlem,Boylam,Giriş,Çıkış,Süre(sn),Oran(%)\n';
  done.forEach((s,i)=>{
    const dur=Math.round((s.endTime-s.startTime)/1000);
    const pct=(((s.endTime-s.startTime)/grand)*100).toFixed(1);
    const addr=(s.address||'').replace(/"/g,"'");
    csv+=`${i+1},"${addr}",${s.lat.toFixed(6)},${s.lng.toFixed(6)},`+
         `${new Date(s.startTime).toISOString()},${new Date(s.endTime).toISOString()},${dur},%${pct}\n`;
  });
  const blob=new Blob(['\uFEFF'+csv],{type:'text/csv;charset=utf-8;'});
  const a=document.createElement('a');
  a.href=URL.createObjectURL(blob);
  a.download=`konum-rapor-${new Date().toISOString().slice(0,10)}.csv`;
  a.click(); showInfo('CSV indiriliyor...');
}
